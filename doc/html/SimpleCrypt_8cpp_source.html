64void SimpleCrypt::splitKey()
65{
66 myKeyParts.clear();
67 myKeyParts.resize(8);
68 for (int i = 0;i < 8; i++)
69 {
70 quint64 part = myKey;
71 for (int j = i; j > 0; j--) { part = part >> 8; }
72 part = part & 0xff;
73 myKeyParts[i] = static_cast<char>(part);
74 }

89QByteArray SimpleCrypt::encryptToByteArray(const QByteArray &plaintext)
90{
91 if (myKeyParts.isEmpty())
92 {
93 qWarning() << "No key set.";
94 myLastError = ErrorNoKeySet;
95 return QByteArray();
96 }
97
98 QByteArray ba = plaintext;
99
100 CryptoFlags flags = CryptoFlagNone;
101 if (myCompressionMode == CompressionAlways)
102 {
103 ba = qCompress(ba, 9); // maximum compression
104 flags |= CryptoFlagCompression;
105 }
106 else if (myCompressionMode == CompressionAuto)
107 {
108 QByteArray compressed = qCompress(ba, 9);
109 if (compressed.count() < ba.count())
110 {
111 ba = compressed;
112 flags |= CryptoFlagCompression;
113 }
114 }
115
116 QByteArray integrityProtection;
117 if (myProtectionMode == ProtectionChecksum)
118 {
119 flags |= CryptoFlagChecksum;
120 QDataStream s(&integrityProtection, QIODevice::WriteOnly);
121 s << qChecksum(ba.constData(), ba.length());
122 }
123 else if (myProtectionMode == ProtectionHash)
124 {
125 flags |= CryptoFlagHash;
126 QCryptographicHash hash(QCryptographicHash::Sha1);
127 hash.addData(ba);
128
129 integrityProtection += hash.result();
130 }
131 // prepend a random char to the string
132 QRandomGenerator generator(uint(QDateTime::currentMSecsSinceEpoch() & 0xFFFF));
133 char randomChar = char(generator.generate() & 0xFF);
134 // char randomChar = char(qrand() & 0xFF);
135
136 ba = randomChar + integrityProtection + ba;
137
138 int pos(0);
139 char lastChar(0);
140
141 int cnt = ba.count();
142
143 while (pos < cnt)
144 {
145 ba[pos] = ba.at(pos) ^ myKeyParts.at(pos % 8) ^ lastChar;
146 lastChar = ba.at(pos);
147 ++pos;
148 }
149
150 QByteArray resultArray;
151 resultArray.append(char(0x03)); // version for future updates to algorithm
152 resultArray.append(char(flags)); // encryption flags
153 resultArray.append(ba);
154
155 myLastError = ErrorNoError;
156 return resultArray;
157}

217QByteArray SimpleCrypt::decryptToByteArray(QByteArray cypher)
218{
219 if (myKeyParts.isEmpty())
220 {
221 qWarning() << "No key set.";
222 myLastError = ErrorNoKeySet;
223 return QByteArray();
224 }
225
226 QByteArray ba = cypher;
227
228 if( cypher.count() < 3 )
229 { return QByteArray(); }
230
231 char version = ba.at(0);
232
233 if (version != 3)
234 { // we only work with version 3
235 myLastError = ErrorUnknownVersion;
236 qWarning() << "Invalid version or not a cyphertext.";
237 return QByteArray();
238 }
239
240 CryptoFlags flags = CryptoFlags(ba.at(1));
241
242 ba = ba.mid(2);
243 int pos(0);
244 int cnt(ba.count());
245 char lastChar = 0;
246
247 while (pos < cnt)
248 {
249 char currentChar = ba[pos];
250 ba[pos] = ba.at(pos) ^ lastChar ^ myKeyParts.at(pos % 8);
251 lastChar = currentChar;
252 ++pos;
253 }
254
255 ba = ba.mid(1); // chop off the random number at the start
256
257 bool integrityOk(true);
258 if (flags.testFlag(CryptoFlagChecksum))
259 {
260 if (ba.length() < 2)
261 {
262 myLastError = ErrorIntegrityFailed;
263 return QByteArray();
264 }
265 quint16 storedChecksum;
266 {
267 QDataStream s(&ba, QIODevice::ReadOnly);
268 s >> storedChecksum;
269 }
270 ba = ba.mid(2);
271 quint16 checksum = qChecksum(ba.constData(), ba.length());
272 integrityOk = (checksum == storedChecksum);
273 }
274 else if (flags.testFlag(CryptoFlagHash))
275 {
276 if (ba.length() < 20)
277 {
278 myLastError = ErrorIntegrityFailed;
279 return QByteArray();
280 }
281 QByteArray storedHash = ba.left(20);
282 ba = ba.mid(20);
283 QCryptographicHash hash(QCryptographicHash::Sha1);
284 hash.addData(ba);
285 integrityOk = (hash.result() == storedHash);
286 }
287
288 if (!integrityOk)
289 {
290 myLastError = ErrorIntegrityFailed;
291 return QByteArray();
292 }
293
294 if (flags.testFlag(CryptoFlagCompression))
295 { ba = qUncompress(ba); }
296
297 myLastError = ErrorNoError;
298 return ba;
299}
