521int mpreal_init(PyObject *self, PyObject *args, PyObject *kwds)
522{
523 Q_UNUSED(kwds)
524 PyObject *arg0 = PyNull;
525 PyObject *arg1 = PyNull;
526 if (!PyArg_ParseTuple(args, "|OO:", &arg0, &arg1))
527 return -1;
528
529 if (arg0 != PyNull)
530 {
531 if (PyMpReal_Check(arg0))
532 {
533 ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(PyMpReal_AsMpReal(arg0));
534 }
535 else if (PyFloat_Check(arg0))
536 {
537 ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(PyFloat_AsDouble(arg0));
538 }
539 else if (PyLong_Check(arg0))
540 {
541 ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(PyLong_AsLong(arg0));
542 }
543 else if (PyUnicode_Check(arg0))
544 {
545 int base = 10;
546 if (arg1 != PyNull && PyLong_Check(arg1))
547 {
548 long tmp = PyLong_AsLong(arg1);
549 if (PyErr_Occurred())
550 { return -1; }
551 if (tmp > std::numeric_limits<int>::max() || tmp < std::numeric_limits<int>::min())
552 {
553 PyErr_SetString(PyExc_OverflowError, "cannot init mpreal: base argument overflow.");
554 return -1;
555 }
556 base = (int) tmp;
557 }
558 if (base < 2 || base > 256)
559 {
560 PyErr_SetString(PyExc_RuntimeError, "base argument must be in the range 2 - 256");
561 return -1;
562 }
563 //Check if string can be converted by mpfr and otherwise set error.
564 std::string argStr = PyUnicode_AsUTF8(arg0);
565 mpfr_t x;
566 mpfr_init2(x, 64);
567 int r = mpfr_set_str(x, argStr.c_str(), base, MPFR_RNDN);
568 if (r != 0)
569 {
570 PyErr_SetString(PyExc_ValueError, ("could not convert string to mpreal: '" + argStr + "'").c_str());
571 return -1;
572 }
573 else
574 {
575 ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(argStr, mpfr::mpreal::get_default_prec(), base);
576 }
577 }
578 else
579 {
580 PyErr_BadArgument();
581 return -1;
582 }
583 }
584 else
585 {
586 ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal();
587 }
588
589 return 0;
590}
