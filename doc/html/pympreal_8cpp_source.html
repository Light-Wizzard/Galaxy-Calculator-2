522 int mpreal_init(PyObject *self, PyObject *args, PyObject *kwds)
523 {
524  Q_UNUSED(kwds)
525  PyObject *arg0 = PyNull;
526  PyObject *arg1 = PyNull;
527  if (!PyArg_ParseTuple(args, "|OO:", &arg0, &arg1))
528  return -1;
529 
530  if (arg0 != PyNull)
531  {
532  if (PyMpReal_Check(arg0))
533  {
534  ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(PyMpReal_AsMpReal(arg0));
535  }
536  else if (PyFloat_Check(arg0))
537  {
538  ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(PyFloat_AsDouble(arg0));
539  }
540  else if (PyLong_Check(arg0))
541  {
542  ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(PyLong_AsLong(arg0));
543  }
544  else if (PyUnicode_Check(arg0))
545  {
546  int base = 10;
547  if (arg1 != PyNull && PyLong_Check(arg1))
548  {
549  long tmp = PyLong_AsLong(arg1);
550  if (PyErr_Occurred())
551  { return -1; }
552  if (tmp > std::numeric_limits<int>::max() || tmp < std::numeric_limits<int>::min())
553  {
554  PyErr_SetString(PyExc_OverflowError, "cannot init mpreal: base argument overflow.");
555  return -1;
556  }
557  base = (int) tmp;
558  }
559  if (base < 2 || base > 256)
560  {
561  PyErr_SetString(PyExc_RuntimeError, "base argument must be in the range 2 - 256");
562  return -1;
563  }
564  //Check if string can be converted by mpfr and otherwise set error.
565  std::string argStr = PyUnicode_AsUTF8(arg0);
566  mpfr_t x;
567  mpfr_init2(x, 64);
568  int r = mpfr_set_str(x, argStr.c_str(), base, MPFR_RNDN);
569  if (r != 0)
570  {
571  PyErr_SetString(PyExc_ValueError, ("could not convert string to mpreal: '" + argStr + "'").c_str());
572  return -1;
573  }
574  else
575  {
576  ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal(argStr, mpfr::mpreal::get_default_prec(), base);
577  }
578  }
579  else
580  {
581  PyErr_BadArgument();
582  return -1;
583  }
584  }
585  else
586  {
587  ((PyMpRealObject *) self)->mpreal = new mpfr::mpreal();
588  }
589 
590  return 0;
591 }
